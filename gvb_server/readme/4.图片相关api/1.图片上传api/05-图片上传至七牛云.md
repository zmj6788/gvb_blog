4.使用插件

```
//保存至本地之前判断是否上传至七牛云

    if global.Config.QiNiu.Enable {

      filePath, err := qiniu.UploadImage(byteData, filename, "images")

      if err != nil {

        logrus.Error(err)

        iml = append(iml, FileUploadResponse{

          FileName:  filename,

          IsSuccess: false,

          Msg:       err.Error(),

        })

        continue

      }

      iml = append(iml, FileUploadResponse{

        FileName:  filePath,

        IsSuccess: true,

        Msg:       "上传七牛云成功",

      })

      global.DB.Create(&models.BannerModel{

      Path: filePath,

      Hash: hashstr,

      Name: filename,

      ImageType: ctype.QiNiu,

    })

      continue

    }
```
